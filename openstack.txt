sudo su

echo "stack ALL=(ALL) NOPASSWD: ALL | sudo tee / etc/sudoers.d/stack

sudo su-stack

If git not installed:

sudo apt install git-y

git clone https://github.com/openstack-dev/devs... b stable/pike devstack/

cd devstack/

ifconfig

sudo apt install net-tools
